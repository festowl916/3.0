async function submitForm() { const school = document.getElementById('school').value.trim(); if (!school) { message.innerHTML = `
Isi nama sekolah
`; return; } try { const res = await fetch(API_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ ic: currentIc, school: school }) }); const data = await res.json(); if (data.status === "success") { message.innerHTML = `
${data.message}
`; } else if (data.status === "duplicate") { message.innerHTML = `
${data.message}
`; } else { message.innerHTML = `
Ralat sistem
`; } infoBox.style.display = 'none'; icInput.value = ''; document.getElementById('school').value = ''; } catch (err) { console.log(err); message.innerHTML = `
Ralat sambungan
`; } }
